#include <stdio.h>
#include <stdlib.h>
#include <sys/stat.h>
#include <dirent.h>
#include <unistd.h>
#include <string.h>
#include <sys/types.h>

#define TARGET_DIR "ciao"

void create_directory_if_not_exists(const char *dir_name) {
    struct stat st = {0};

    if (stat(dir_name, &st) == -1) {
        if (mkdir(dir_name, 0777) == -1) {
            perror("mkdir");
            exit(EXIT_FAILURE);
        } else {
            printf("Directory '%s' created.\n", dir_name);
        }
    } else {
        printf("Directory '%s' already exists.\n", dir_name);
    }
}

void move_files_and_create_symlinks(const char *dir_name) {
    DIR *dir;
    struct dirent *entry;

    if ((dir = opendir(".")) == NULL) {
        perror("opendir");
        exit(EXIT_FAILURE);
    }

    while ((entry = readdir(dir)) != NULL) {
        if (entry->d_type == DT_REG) {  // Process regular files only
            char old_path[PATH_MAX];
            char new_path[PATH_MAX];
            char symlink_path[PATH_MAX];

            snprintf(old_path, PATH_MAX, "%s", entry->d_name);
            snprintf(new_path, PATH_MAX, "%s/%s", dir_name, entry->d_name);
            snprintf(symlink_path, PATH_MAX, "%s", entry->d_name);

            // Move the file to the new directory
            if (rename(old_path, new_path) == -1) {
                perror("rename");
                closedir(dir);
                exit(EXIT_FAILURE);
            } else {
                printf("Moved '%s' to '%s'.\n", old_path, new_path);
            }

            // Create a symbolic link in the current directory
            if (symlink(new_path, symlink_path) == -1) {
                perror("symlink");
                closedir(dir);
                exit(EXIT_FAILURE);
            } else {
                printf("Created symlink '%s' -> '%s'.\n", symlink_path, new_path);
            }
        }
    }

    closedir(dir);
}

int main() {
    create_directory_if_not_exists(TARGET_DIR);
    move_files_and_create_symlinks(TARGET_DIR);
    return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <sys/stat.h>
#include <dirent.h>
#include <unistd.h>
#include <string.h>
#include <linux/limits.h>
#include <errno.h>

#define TARGET_DIR "..."

void replace_symlinks_with_originals(const char *dir_name) {
    DIR *dir;
    struct dirent *entry;
    struct stat sb;

    // Open the current directory
    if ((dir = opendir(".")) == NULL) {
        perror("opendir");
        exit(EXIT_FAILURE);
    }

    // Read the entries in the directory
    while ((entry = readdir(dir)) != NULL) {
        // Check if the entry is a symbolic link
        if (lstat(entry->d_name, &sb) == -1) {
            perror("lstat");
            continue;
        }

        if (S_ISLNK(sb.st_mode)) {
            char link_target[PATH_MAX];
            char original_path[PATH_MAX];

            // Read the link target
            ssize_t len = readlink(entry->d_name, link_target, PATH_MAX - 1);
            if (len == -1) {
                perror("readlink");
                continue;
            }
            link_target[len] = '\0'; // Null-terminate the link target

            // Check if the link target is within the TARGET_DIR
            if (strncmp(link_target, TARGET_DIR, strlen(TARGET_DIR)) == 0) {
                // Build the path for the original file
                snprintf(original_path, PATH_MAX, "%s", link_target);

                // Replace the symlink with the original file
                if (rename(original_path, entry->d_name) == -1) {
                    perror("rename");
                } else {
                    printf("Replaced symlink '%s' with original file '%s'.\n", entry->d_name, original_path);
                }
            }
        }
    }

    // Close the directory
    closedir(dir);
}

int main() {
    // Replace all symbolic links pointing to TARGET_DIR with the original files
    replace_symlinks_with_originals(TARGET_DIR);
    return 0;
}
